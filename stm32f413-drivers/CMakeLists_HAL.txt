cmake_minimum_required(VERSION 3.24)

project(HAL)

# Not recomended to use GLOB but this allows us to flexibly change HAL source
add_library(HAL)

#Sources
file(GLOB CMSIS_DEVICE_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/Device/Source/*.c)
file(GLOB CMSIS_DSP_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/Source/*.c)
file(GLOB CMSIS_NN_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/NN/Source/*.c)
if(PROCESSOR_FAM = F4XX)
    file(GLOB CMSIS_DSP_LIB_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/ComputeLibrary/Source/*.c)
    file(GLOB DAP_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DAP/Source/*.c)
    file(GLOB HAL_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/STM32F4xx_HAL_Driver/Src/*.c)
else
    file(GLOB CMSIS_DSP_LIB_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/DSP_Lib_TestSuite/RefLibs/src/*.c)
    file(GLOB DSP_COMMON_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/DSP_Lib_TestSuite/Common/src/*.c)
    file(GLOB HAL_SOURCES CONFIGURE_DEPENDS ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/STM32H7xx_HAL_Driver/Src/*.c)
endif()

file(GLOB SOURCES CONFIGURE_DEPENDS ${PROCESSOR_FAM}/Src/*.c)
file(GLOB SOURCES_CMSIS CONFIGURE_DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/../CMSIS/*.c)


target_sources(HAL PUBLIC
${SOURCES}
${SOURCES_CMSIS}
${CMAKE_CURRENT_SOURCE_DIR}/../CMSIS/system_stm32f4xx.c
)


target_include_directories(HAL PUBLIC
    ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/Include
    ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/Core/Include
    ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/Device/Include
    ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/Include
    ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/NN/Include

    if(PROCESSOR_FAM=F4XX)
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DAP/Include
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/STM32F4xx_HAL_Driver/Inc
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/ComputeLibrary/Include
    else
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/DSP_Lib_TestSuite/Common/inc
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/CMSIS/DSP/DSP_Lib_TestSuite/RefLibs/inc
        ${CMAKE_SOURCE_DIR}/${PROCESSOR_FAM}/STM32H7xx_HAL_Driver/Inc
    endif()


    ${CMAKE_SOURCE_DIR}/Inc #stm32f4xx_hal_conf.h
    ${PROCESSOR_FAM}/Inc
    ${PROCESSOR_FAM}/Inc/Legacy
    ${CMAKE_CURRENT_SOURCE_DIR}/../CMSIS
    ${CMAKE_CURRENT_SOURCE_DIR}/../CMSIS/Include
    ${CMAKE_CURRENT_SOURCE_DIR}/../CMSIS/Device/${PROCESSOR})
