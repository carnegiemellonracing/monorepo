#F4XX HAL Sources
file(GLOB HAL_SOURCES_F4XX CONFIGURE_DEPENDS F4XX/STM32F4xx_HAL_Driver/Src/*.c)

#H7XX CMSIS Sources
file(GLOB H7XX_CMSIS_DEVICE_SOURCES CONFIGURE_DEPENDS H7XX/CMSIS/Device/Source/*.c)

#H7XX HAL Sources

file(GLOB SOURCES CONFIGURE_DEPENDS ${PROCESSOR_FAM}/Src/*.c)

target_sources(HAL PUBLIC

)

    H7XX/CMSIS/DSP/Include
    H7XX/CMSIS/DSP/DSP_Lib_TestSuite/Common/inc
    H7XX/CMSIS/DSP/DSP_Lib_TestSuite/Common/JTest/inc
    H7XX/CMSIS/DSP/DSP_Lib_TestSuite/RefLibs/inc
    H7XX/CMSIS/NN/Include
    H7XX/STM32H7xx_HAL_Driver/Inc


    ${CMAKE_SOURCE_DIR}/Inc #stm32f4xx_hal_conf.h
